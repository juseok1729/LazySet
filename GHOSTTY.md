# ghostty 설정

```
# 커서 스타일
cursor-style = block

# 커서 깜빡임 활성화
cursor-style-blink = true

# 커서 불투명(0=투명, 1=불투명)
cursor-opacity = 1

# 뒷 배경 투명도 (0=투명, 1=불투명)
background-opacity = 0.8

# 뒷 배경 흐림 활성화
background-blur = true

# 맥 에뮬레이터 타이틀바 제거
macos-titlebar-style = hidden

# 창, pane 등을 닫을 때 확인창을 표시하지 않음.
confirm-close-surface = false

# 윈도우 패딩 (좌/우)
window-padding-x = 5,5

# 윈도우 패딩 (상/하)
window-padding-y = 5,5
```
